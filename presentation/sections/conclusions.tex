\section{Conclusions}
\begin{frame}{Conclusions}
    \begin{block}{Achievements}
        \begin{itemize}
            \item Successful implementation of \textbf{TE} and \textbf{CJTE} using the \textbf{DVP} technique.
            \item Characterized information flow in both synthetic and physiological systems.
            \item Provided a robust, parameter-free framework for causal analysis.
        \end{itemize}
    \end{block}

    \begin{block}{Key Methodological Insights}
        \begin{itemize}
            \item \textbf{Complementary Usage:} CJTE excels in multivariate environments but should be used alongside TE to avoid \textit{synergistic inflation}.
            \item \textbf{Tool Sensitivity:} Both metrics are highly effective at detecting \textbf{directionality} and \textbf{nonlinearity} in complex regulatory systems.
        \end{itemize}
    \end{block}

    \begin{block}{Future Work}
      \begin{itemize}
          \item Investigation of bias-correction and hyperparameter tuning.
          \item Validation on larger, diverse cohorts.
      \end{itemize}
    \end{block}
\end{frame}
